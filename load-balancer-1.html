function calculateTotal(items) {
  let total = 0;
  for (const item of items) {
    // Ensure item price is a valid number before adding
    if (typeof item.price === 'number' && !isNaN(item.price)) {
       total += item.price;
    } else {
       console.warn(`Invalid price found for item: ${item.name}`);
    }
  }
  return total;
}
